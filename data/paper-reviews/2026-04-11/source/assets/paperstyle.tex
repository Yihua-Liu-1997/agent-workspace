    \shade[
      shading=axis,
      shading angle=72,
      left color=\badgeedgecolor{#1},
      middle color=\badgemidcolor{#1},
      right color=\badgelightcolor{#1},
      draw=\badgeedgecolor{#1},
      line width=0.04pt

  }%
